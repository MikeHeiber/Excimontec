121  // pre-calculated distances vector that contains the distances to nearby sites used for event execution time calculations
122  std::vector<double> distances;
123  // pre-calculated isInDissRange and isInFRETRange vectors that contains booleans to indicate whether the nearby sites are within range for the different exciton events to be possible.
124  std::vector<bool> isInDissRange;

129  ExcitonEventCalcVars(OSC_Sim* sim_ptr) {
130  range = (int)ceil(((sim_ptr->params.FRET_cutoff > sim_ptr->params.Exciton_dissociation_cutoff) ? (sim_ptr->params.FRET_cutoff) : (sim_ptr->params.Exciton_dissociation_cutoff)) / sim_ptr->lattice.getUnitSize());
131  dim = (2 * range + 1);
132  hop_event = Exciton_Hop(sim_ptr);
133  hops_temp.assign(dim*dim*dim, hop_event);
134  diss_event = Exciton_Dissociation(sim_ptr);
135  dissociations_temp.assign(dim*dim*dim, diss_event);
136  ee_annihilation_event = Exciton_Exciton_Annihilation(sim_ptr);
137  ee_annihilations_temp.assign(dim*dim*dim, ee_annihilation_event);
138  ep_annihilation_event = Exciton_Polaron_Annihilation(sim_ptr);
139  ep_annihilations_temp.assign(dim*dim*dim, ep_annihilation_event);
140  hops_valid.assign(dim*dim*dim, false);
141  dissociations_valid.assign(dim*dim*dim, false);
142  ee_annihilations_valid.assign(dim*dim*dim, false);
143  ep_annihilations_valid.assign(dim*dim*dim, false);
144  // pre-calculated distances vector that contains the distances to nearby sites used for event execution time calculations
145  distances.assign(dim*dim*dim, 0.0);
146  // pre-calculated isInDissRange and isInFRETRange vectors that contains booleans to indicate whether the nearby sites are within range for the different exciton events to be possible.
147  isInDissRange.assign(dim*dim*dim, false);
148  isInFRETRange.assign(dim*dim*dim, false);
149  // Initiaize distances, isInDissRange, and isInFRETRange vectors
150  for (int i = -range; i <= range; i++) {
151  for (int j = -range; j <= range; j++) {
152  for (int k = -range; k <= range; k++) {
153  int index = (i + range)*dim*dim + (j + range)*dim + (k + range);
154  distances[index] = sim_ptr->lattice.getUnitSize()*sqrt((double)(i*i + j * j + k * k));
155  if (!((distances[index] - 0.0001) > sim_ptr->params.Exciton_dissociation_cutoff)) {
156  isInDissRange[index] = true;
157  }
158  if (!((distances[index] - 0.0001) > sim_ptr->params.FRET_cutoff)) {
159  isInFRETRange[index] = true;
160  }
161  }
162  }
163  }
164  }
165  };

185  PolaronEventCalcVars(OSC_Sim* sim_ptr) {
186  range = (int)ceil(sim_ptr->params.Polaron_hopping_cutoff / sim_ptr->lattice.getUnitSize());
187  dim = (2 * range + 1);
188  hop_event = Polaron_Hop(sim_ptr);
189  hops_temp.assign(dim*dim*dim, hop_event);
190  rec_event = Polaron_Recombination(sim_ptr);
191  recombinations_temp.assign(dim*dim*dim, rec_event);
192  hops_valid.assign(dim*dim*dim, false);
193  recombinations_valid.assign(dim*dim*dim, false);
194  // pre-calculated distances vector that contains the distances to nearby sites used for event execution time calculations
195  distances.assign(dim*dim*dim, 0.0);
196  E_deltas.assign(dim*dim*dim, 0.0);
197  // pre-calculated isInRange vector that contains booleans to indicate if the nearby sites are within range for polaron events to be possible.
198  isInRange.assign(dim*dim*dim, false);
199  // Intialize distances and isInRange vectors
200  for (int i = -range; i <= range; i++) {
201  for (int j = -range; j <= range; j++) {
202  for (int k = -range; k <= range; k++) {
203  int index = (i + range)*dim*dim + (j + range)*dim + (k + range);
204  distances[index] = sim_ptr->lattice.getUnitSize()*sqrt((double)(i*i + j * j + k * k));
205  if (!((distances[index] - 0.0001) > sim_ptr->params.Polaron_hopping_cutoff)) {
206  isInRange[index] = true;
207  }
208  }
209  }
210  }
211  }
