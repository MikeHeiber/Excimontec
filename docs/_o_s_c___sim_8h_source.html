197  struct ExcitonEventCalcVars {
198  int range;
199  int dim;
200  Exciton_Hop hop_event;
201  std::vector<Exciton_Hop> hops_temp;
202  Exciton_Dissociation diss_event;
203  std::vector<Exciton_Dissociation> dissociations_temp;
204  Exciton_Exciton_Annihilation ee_annihilation_event;
205  std::vector<Exciton_Exciton_Annihilation> ee_annihilations_temp;
206  Exciton_Polaron_Annihilation ep_annihilation_event;
207  std::vector<Exciton_Polaron_Annihilation> ep_annihilations_temp;
208  std::vector<bool> hops_valid;
209  std::vector<bool> dissociations_valid;
210  std::vector<bool> ee_annihilations_valid;
211  std::vector<bool> ep_annihilations_valid;
212  // pre-calculated distances vector that contains the distances to nearby sites used for event execution time calculations
213  std::vector<double> distances;
214  // pre-calculated isInDissRange and isInFRETRange vectors that contains booleans to indicate whether the nearby sites are within range for the different exciton events to be possible.
215  std::vector<bool> isInDissRange;
216  std::vector<bool> isInFRETRange;
217 
218  ExcitonEventCalcVars() {}
219 
220  ExcitonEventCalcVars(OSC_Sim* sim_ptr) {
221  range = (int)ceil(((sim_ptr->FRET_cutoff > sim_ptr->Exciton_dissociation_cutoff) ? (sim_ptr->FRET_cutoff) : (sim_ptr->Exciton_dissociation_cutoff)) / sim_ptr->lattice.getUnitSize());
222  dim = (2 * range + 1);
223  hop_event = Exciton_Hop(sim_ptr);
224  hops_temp.assign(dim*dim*dim, hop_event);
225  diss_event = Exciton_Dissociation(sim_ptr);
226  dissociations_temp.assign(dim*dim*dim, diss_event);
227  ee_annihilation_event = Exciton_Exciton_Annihilation(sim_ptr);
228  ee_annihilations_temp.assign(dim*dim*dim, ee_annihilation_event);
229  ep_annihilation_event = Exciton_Polaron_Annihilation(sim_ptr);
230  ep_annihilations_temp.assign(dim*dim*dim, ep_annihilation_event);
231  hops_valid.assign(dim*dim*dim, false);
232  dissociations_valid.assign(dim*dim*dim, false);
233  ee_annihilations_valid.assign(dim*dim*dim, false);
234  ep_annihilations_valid.assign(dim*dim*dim, false);
235  // pre-calculated distances vector that contains the distances to nearby sites used for event execution time calculations
236  distances.assign(dim*dim*dim, 0.0);
237  // pre-calculated isInDissRange and isInFRETRange vectors that contains booleans to indicate whether the nearby sites are within range for the different exciton events to be possible.
238  isInDissRange.assign(dim*dim*dim, false);
239  isInFRETRange.assign(dim*dim*dim, false);
240  // Initiaize distances, isInDissRange, and isInFRETRange vectors
241  for (int i = -range; i <= range; i++) {
242  for (int j = -range; j <= range; j++) {
243  for (int k = -range; k <= range; k++) {
244  int index = (i + range)*dim*dim + (j + range)*dim + (k + range);
245  distances[index] = sim_ptr->lattice.getUnitSize()*sqrt((double)(i*i + j * j + k * k));
246  if (!((distances[index] - 0.0001) > sim_ptr->Exciton_dissociation_cutoff)) {
247  isInDissRange[index] = true;
248  }
249  if (!((distances[index] - 0.0001) > sim_ptr->FRET_cutoff)) {
250  isInFRETRange[index] = true;
251  }
252  }
253  }
254  }
255  }
256  };

276  PolaronEventCalcVars(OSC_Sim* sim_ptr) {
277  range = (int)ceil(sim_ptr->Polaron_hopping_cutoff / sim_ptr->lattice.getUnitSize());
278  dim = (2 * range + 1);
279  hop_event = Polaron_Hop(sim_ptr);
280  hops_temp.assign(dim*dim*dim, hop_event);
281  rec_event = Polaron_Recombination(sim_ptr);
282  recombinations_temp.assign(dim*dim*dim, rec_event);
283  hops_valid.assign(dim*dim*dim, false);
284  recombinations_valid.assign(dim*dim*dim, false);
285  // pre-calculated distances vector that contains the distances to nearby sites used for event execution time calculations
286  distances.assign(dim*dim*dim, 0.0);
287  E_deltas.assign(dim*dim*dim, 0.0);
288  // pre-calculated isInRange vector that contains booleans to indicate if the nearby sites are within range for polaron events to be possible.
289  isInRange.assign(dim*dim*dim, false);
290  // Intialize distances and isInRange vectors
291  for (int i = -range; i <= range; i++) {
292  for (int j = -range; j <= range; j++) {
293  for (int k = -range; k <= range; k++) {
294  int index = (i + range)*dim*dim + (j + range)*dim + (k + range);
295  distances[index] = sim_ptr->lattice.getUnitSize()*sqrt((double)(i*i + j * j + k * k));
296  if (!((distances[index] - 0.0001) > sim_ptr->Polaron_hopping_cutoff)) {
297  isInRange[index] = true;
298  }
299  }
300  }
301  }
302  }
303  };
