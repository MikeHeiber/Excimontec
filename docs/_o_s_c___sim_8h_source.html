384  std::vector<bool> ep_annihilations_valid;
385  // precalculated distances vector that contains the distances to nearby sites used for event execution time calculations
386  std::vector<double> distances;
387  // precalculated isInDissRange and isInFRETRange vectors that contains booleans to indicate whether the nearby sites are within range for the different exciton events to be possible.
388  std::vector<bool> isInDissRange;

393  ExcitonEventCalcVars(OSC_Sim* sim_ptr) {
394  range = (int)ceil(((sim_ptr->params.FRET_cutoff > sim_ptr->params.Exciton_dissociation_cutoff) ? (sim_ptr->params.FRET_cutoff) : (sim_ptr->params.Exciton_dissociation_cutoff)) / sim_ptr->lattice.getUnitSize());
395  dim = (2 * range + 1);
396  hop_event = Exciton::Hop(sim_ptr);
397  hops_temp.assign(dim*dim*dim, hop_event);
398  diss_event = Exciton::Dissociation(sim_ptr);
399  dissociations_temp.assign(dim*dim*dim, diss_event);
400  ee_annihilation_event = Exciton::Exciton_Annihilation(sim_ptr);
401  ee_annihilations_temp.assign(dim*dim*dim, ee_annihilation_event);
402  ep_annihilation_event = Exciton::Polaron_Annihilation(sim_ptr);
403  ep_annihilations_temp.assign(dim*dim*dim, ep_annihilation_event);
404  hops_valid.assign(dim*dim*dim, false);
405  dissociations_valid.assign(dim*dim*dim, false);
406  ee_annihilations_valid.assign(dim*dim*dim, false);
407  ep_annihilations_valid.assign(dim*dim*dim, false);
408  // precalculated distances vector that contains the distances to nearby sites used for event execution time calculations
409  distances.assign(dim*dim*dim, 0.0);
410  // precalculated isInDissRange and isInFRETRange vectors that contains booleans to indicate whether the nearby sites are within range for the different exciton events to be possible.
411  isInDissRange.assign(dim*dim*dim, false);
412  isInFRETRange.assign(dim*dim*dim, false);
413  // Initialize distances, isInDissRange, and isInFRETRange vectors
414  for (int i = -range; i <= range; i++) {
415  for (int j = -range; j <= range; j++) {
416  for (int k = -range; k <= range; k++) {
417  int index = (i + range)*dim*dim + (j + range)*dim + (k + range);
418  distances[index] = sim_ptr->lattice.getUnitSize()*sqrt((double)(i*i + j * j + k * k));
419  if (!((distances[index] - 0.0001) > sim_ptr->params.Exciton_dissociation_cutoff)) {
420  isInDissRange[index] = true;
421  }
422  if (!((distances[index] - 0.0001) > sim_ptr->params.FRET_cutoff)) {
423  isInFRETRange[index] = true;
424  }
425  }
426  }
427  }
428  }
429  };

449  PolaronEventCalcVars(OSC_Sim* sim_ptr) {

451  dim = (2 * range + 1);

455  recombinations_temp.assign(dim*dim*dim, rec_event);
456  hops_valid.assign(dim*dim*dim, false);
457  recombinations_valid.assign(dim*dim*dim, false);
458  // precalculated distances vector that contains the distances to nearby sites used for event execution time calculations
459  distances.assign(dim*dim*dim, 0.0);
460  E_deltas.assign(dim*dim*dim, 0.0);
461  // precalculated isInRange vector that contains booleans to indicate if the nearby sites are within range for polaron events to be possible.
462  isInRange.assign(dim*dim*dim, false);
463  // Initialize distances and isInRange vectors
464  for (int i = -range; i <= range; i++) {
465  for (int j = -range; j <= range; j++) {
466  for (int k = -range; k <= range; k++) {
467  int index = (i + range)*dim*dim + (j + range)*dim + (k + range);
468  distances[index] = sim_ptr->lattice.getUnitSize()*sqrt((double)(i*i + j * j + k * k));
469  if (!((distances[index] - 0.0001) > sim_ptr->params.Polaron_hopping_cutoff)) {
470  isInRange[index] = true;
471  }
472  }
473  }
474  }
