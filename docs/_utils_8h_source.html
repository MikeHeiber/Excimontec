375  double result;
376  if (exponent == 0) {

422  void removeDuplicates(std::vector<T> &vec) {
423  if ((int)vec.size() > 2) {
424  for (auto it1 = vec.begin(); it1 != vec.end() - 2; ++it1) {
425  for (auto it2 = it1 + 2; it2 != vec.end(); ++it2) {
426  if (*it2 == *it1) {
427  *it2 = *(it2 - 1);
428  }
429  }
430  }
431  }
432  if ((int)vec.size() > 1) {
433  auto it3 = unique(vec.begin(), vec.end());
434  vec.resize(distance(vec.begin(), it3));

453  template<typename T, typename A>
454  double vector_median(std::vector<T, A> data) {
455  if (data.size() % 2 == 0) {
456  const auto median_it1 = data.begin() + data.size() / 2 - 1;
457  const auto median_it2 = data.begin() + data.size() / 2;
458  std::nth_element(data.begin(), median_it1, data.end());
459  T val1 = *median_it1;
460  std::nth_element(data.begin(), median_it2, data.end());
461  T val2 = *median_it2;
462  return (double)(val1 + val2) / 2.0;
463  }
464  else {
465  const auto median_it = data.begin() + data.size() / 2;
466  std::nth_element(data.begin(), median_it, data.end());
467  return *median_it;
468  }
469  }
