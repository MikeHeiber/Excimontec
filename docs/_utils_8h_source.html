327  double array_stdev(const T data[], const int array_size) {
328  double sum = 0;

358  double result;
359  if (exponent == 0) {
360  result = 1.0;
361  return result;
362  }
363  result = base;

404  template<typename T>
405  void removeDuplicates(std::vector<T> &vec) {
406  if ((int)vec.size() > 2) {
407  for (auto it1 = vec.begin(); it1 != vec.end() - 2; ++it1) {
408  for (auto it2 = it1 + 2; it2 != vec.end(); ++it2) {
409  if (*it2 == *it1) {
410  *it2 = *(it2 - 1);
411  }
412  }
413  }
414  }
415  if ((int)vec.size() > 1) {
416  auto it3 = unique(vec.begin(), vec.end());
417  vec.resize(distance(vec.begin(), it3));
418  }
419  }

436  template<typename T, typename A>
437  double vector_median(std::vector<T, A> data) {
438  if (data.size() % 2 == 0) {
439  const auto median_it1 = data.begin() + data.size() / 2 - 1;
440  const auto median_it2 = data.begin() + data.size() / 2;
441  std::nth_element(data.begin(), median_it1, data.end());
442  T val1 = *median_it1;
443  std::nth_element(data.begin(), median_it2, data.end());
444  T val2 = *median_it2;
445  return (double)(val1 + val2) / 2.0;
446  }
447  else {
448  const auto median_it = data.begin() + data.size() / 2;
449  std::nth_element(data.begin(), median_it, data.end());
450  return *median_it;
451  }
452  }
